```sql
-- 选择所有列的数据
select * from `${table}`;
-- 从表中选择某一列
select `${colName}` from `${table}`;

-- 当row数据重复时，只列出一条
select distinct `${colName}` from `${table}`;

-- 限制查询多少条
select `${colName}` from `${table}` limit 0, 2;

-- 排序数据 order by 字句优先级高
select `${colname}` from `${table}` order by `${colname}` [limit 0, 2];

-- 可通过制定DESC关键字指定为倒序（默认为升序）；其他未指定的列则会在colname的相同值内进行升序排序
select `${colname}`, `${colname1}` from `${table}` order by `${colname}` desc, `${colname1}`;

-- 使用where字句过滤数据
select `${colname}` from `${table}` where `${colname}`='xxx';

-- 不匹配
select `${colname}` from `${table}` where `${colname}`!='xxx';

-- 空值
select `${colname}` from `${table}` where `${colname}` is NULL;

-- 范围
select `${colname}` from `${table}` where `${colname}` between a and b;

-- 使用and操作符，获取where字句中列数据的交集
select `${colname1}`, `${colname2}`,`${colname3}` from `${table}` where `${colname1}` === 'xxx' and `${colnam2}` > 'xxx';

-- 使用 or 操作符，获取where字句中列数据的并集
select `${colname1}`, `${colname2}`,`${colname3}` from `${table}` where `${colname1}` === 'xxx' or `${colnam2}` > 'xxx';

-- IN操作符用来指定条件范围，范围中的每个条件都可以进行匹配.in取合法值的由逗号分隔的清单，全都括在圆括号中
select `${colname}` from `${table}` where `${colname}` in (1002, 1003);
-- 在使用长的合法选项清单时，IN操作符的语法更清楚且更直观。
--  在使用IN时，计算的次序更容易管理（因为使用的操作符更少）。
--  IN操作符一般比OR操作符清单执行更快。
--  IN的最大优点是可以包含其他SELECT语句，使得能够更动态地建
-- 立WHERE子句。

-- not 配合 in 排除条件范围内的数据
select `${colname}` from `${table}` where `${colname}` not in (1002, 1003);

-- 使用通配符like过滤， “ % ”告诉mysql接收xxx后的任意字符。
select `${colname}` from `${table}` where `${colname}` like 'xxx%';

-- 使用正则表达式搜索
select `${colname}` from `${table}` where `${colname}` regexp '^test';
-- 正则中的or
select `${colname}` from `${table}` where `${colname}` regexp '1$|2$';
-- 匹配几个字符之一
select `${colname}` from `${table}` where `${colname}` regexp '[123] Ton';
```
