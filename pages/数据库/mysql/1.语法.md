```sql
-- 选择所有列的数据
select * from `${table}`;
-- 从表中选择某一列
select `${colName}` from `${table}`;

-- 当row数据重复时，只列出一条
select distinct `${colName}` from `${table}`;

-- 限制查询多少条
select `${colName}` from `${table}` limit 0, 2;

-- 排序数据 order by 字句优先级高
select `${colname}` from `${table}` order by `${colname}` [limit 0, 2];

-- 可通过制定DESC关键字指定为倒序（默认为升序）；其他未指定的列则会在colname的相同值内进行升序排序
select `${colname}`, `${colname1}` from `${table}` order by `${colname}` desc, `${colname1}`;

-- 使用where字句过滤数据
select `${colname}` from `${table}` where `${colname}`='xxx';

-- 使用通配符like过滤， “ % ”告诉mysql接收xxx后的任意字符。
select `${colname}` from `${table}` where `${colname}` like 'xxx%';
```
